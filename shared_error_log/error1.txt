create or alter view RAVEN.VW_LOG_LIST_EXTERNAL_FILE(
	RAVEN_COBID,
	STAGE_NAME,
	STAGE_URL,
	CONTAINER_NAME,
	FOLDER_PATTERN,
	FILE_EXTENSION,
	PROCESS_TIMESTAMP,
	FILE_URL,
	EXTERNAL_FILE_PATH,
	FILE_NAME,
	BLOB_LAST_MODIFIED
) as (
    select
  raven_cobid,
  stage_name,
  stage_url,
  case
      when stage_url like '%azure%'
          then
              case
                  when LEFT(CURRENT_DATABASE(), 4) = 'PROD'
                      then SPLIT_PART(stage_url, '/', -1)
                  else REPLACE(SPLIT_PART(stage_url, '/', -1), SPLIT_PART(SPLIT_PART(stage_url, '/', -1), '-', 1) || '-', '')
              end
      else ''
  end as container_name,
  folder_pattern,
  file_extension,
  process_timestamp,
  value:name::VARCHAR as file_url,
  REPLACE(REPLACE(value:name::VARCHAR, stage_url, ''), stage_name, '') as external_file_path,
  SPLIT_PART(REPLACE(REPLACE(value:name::VARCHAR, stage_url, ''), stage_name, ''), '/', -1) as file_name,
  TO_TIMESTAMP_NTZ(value:last_modified::STRING, 'DY, DD MON YYYY HH24:MI:SS GMT') as blob_last_modified
FROM raven.log_list_external_file as l, lateral FLATTEN(input => l.file_list)
where flag_last_version = true
  );
